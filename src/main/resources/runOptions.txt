# JVM Flags for Assembly Printing
# Add these in IntelliJ VM Options (Run > Edit Configurations > VM Options)
# Requires hsdis-amd64.dll in JDK bin folder for PrintAssembly to work

-XX:+UnlockDiagnosticVMOptions          # required to enable diagnostic flags like PrintAssembly
-XX:+PrintAssembly                      # print C2-compiled native assembly to stderr
-XX:PrintAssemblyOptions=intel          # use Intel syntax instead of AT&T
-XX:-TieredCompilation                  # skip C1, go straight to C2 so output is clean
-XX:CompileThreshold=1                  # compile after 1 invocation, no warmup needed

# print assembly for a specific method
# use dots for the package path, :: before the method name
-XX:CompileCommand=print,logscanner.version04.Tester::<methodName>
